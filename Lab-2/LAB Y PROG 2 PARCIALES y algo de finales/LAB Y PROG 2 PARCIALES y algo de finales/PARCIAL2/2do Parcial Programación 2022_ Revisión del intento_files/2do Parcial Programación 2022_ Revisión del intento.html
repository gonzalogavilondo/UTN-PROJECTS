c)

void preorder(nodoArbol * arbol)

{

if(arbol != NULL)

 {

   printf("%d ", arbol->dato);

   preorder(arbol->izq);

   preorder(arbol->der);

   }

}



void inorder(nodoArbol * arbol)

{

if(arbol != NULL)

 {

   inorder(arbol->izq);

   printf("%d ", arbol->dato);

   inorder(arbol->der);

   }

}



void postorder(nodoArbol * arbol)

{

if(arbol != NULL)

   {

    postorder(arbol->izq);

    postorder(arbol->der);

    printf("%d ", arbol->dato);

    }

}


Retroalimentación

Respuestas:

  1. Hay 3 algoritmos de recorrido: preOrder, inOrder y postOrder.

  2. preOrder: Comienza por el nodo raíz, luego por la izquierda y al final por la derecha.

          inOrder: Comienza por la parte izquierda, luego por el nodo raíz y al final por la derecha.

          posOrder: Comienza por la parte izquierda, luego por la derecha y al final por el nodo raíz.
